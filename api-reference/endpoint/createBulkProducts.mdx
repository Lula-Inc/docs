---
title: "Create Bulk Products"
api: "POST /companies/:companyId/products/bulk"
description: "Create multiple products for a company in a single request."
---

Creates multiple products for the specified company.

### Endpoint

`POST /companies/:companyId/products/bulk`

---

### Path Parameters

<ParamField path="companyId" type="string" required>
  The unique identifier of the company.
</ParamField>

---

### Request Body

<ParamField body="products" type="array of objects" required>
  An array of product objects to create or update.
</ParamField>

### Product Object Fields

<ResponseField name="products" type="array of objects">

<Expandable title="Toggle product object fields">

<ResponseField name="external_id" type="string" required>
  Product identification in your internal system.
</ResponseField>

<ResponseField name="upc" type="string" required>
  Product UPC.
</ResponseField>

<ResponseField name="name" type="string" required>
  Product name.
</ResponseField>

<ResponseField name="size" type="string" required>
  Product size.
</ResponseField>

<ResponseField name="quantity" type="number" required>
  Product quantity (must be a positive number).
</ResponseField>

<ResponseField name="price" type="number" required>
  Product price (minimum 0).
</ResponseField>

<ResponseField name="description" type="string" required>
  Product description.
</ResponseField>

<ResponseField name="image_url" type="string" required>
  Product image URL.
</ResponseField>

<ResponseField name="category" type="string" required>
  Product category.
</ResponseField>

<ResponseField name="brand" type="string">
  Product brand (optional).
</ResponseField>

<ResponseField name="location" type="string">
  Product location (optional).
</ResponseField>

<ResponseField name="unit_count" type="string" required>
  Product unit count.
</ResponseField>

</Expandable>

</ResponseField>

---
